vector<int>res(2,-1);
int left=0,right=nums.size()-1;
//确定左边界
while(left<=right)  
{
    int mid=left+(right-left)/2;
    if (nums[mid] < target) 
    {
        left = mid + 1;
    } 
    else if (nums[mid] > target) 
    {
        right = mid - 1;
    } 
    else if (nums[mid] == target) 
    {
        // 别返回，锁定左侧边界
        right = mid - 1;
    }
}
//左边界已经超出范围
if(left==nums.size())
   return res;
if(nums[left]==target)
   res[0]=left; //确定左边界


left=0,right=nums.size()-1;
 //确定右边界
while(left<=right)   
{
    int mid=left+(right-left)/2;
    if(nums[mid]<target)
         left=mid+1;
    else if(nums[mid]>target)
         right =mid-1;
    else if(nums[mid]==target)
         left=mid+1; // 别返回，锁定右侧边界
}
    
if(right<0)
    return res;
if(nums[right]==target)
    res[1]=right;
